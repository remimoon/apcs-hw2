MERGE SORT

mergesort (deck) {
if (size(deck) == 1) {
return deck;
} else {
 	A = first half of deck;
	B = second half of deck;
	R1 = mergesort (A);
	R2 = mergesort (B);
	merge (R1, R2);
}
}
